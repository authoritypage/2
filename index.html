// Define a function for a simple AI response
function generateResponse(prompt) {
  if (prompt.includes("hello")) {
    return "Hello there! How can I assist you today?";
  } else if (prompt.includes("weather")) {
    return "I'm sorry, I cannot provide real-time weather information.";
  } else {
    return "I am an AI. Please ask me a question.";
  }
}

// Example usage:
let userPrompt = "Hello, AI!";
console.log(generateResponse(userPrompt)); // Output: "Hello there! How can I assist you today?"
